@RestController
@RequestMapping("/api/milestones")
public class MilestoneController {

    @Autowired
    private MilestoneService service;

    @PostMapping
    public ResponseEntity<Milestone> create(@RequestBody Milestone m) {
        return ResponseEntity.ok(service.save(m));
    }
}
